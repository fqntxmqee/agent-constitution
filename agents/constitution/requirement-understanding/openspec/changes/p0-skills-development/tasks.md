### Task-002: 初始化共享类型定义

**描述:** 在 `shared/types.ts` 中定义通用类型

**执行内容:**
```typescript
// 任务类型
export type TaskType = 'development' | 'content' | 'skill' | 'ops' | 'other';
export type TaskStatus = 'pending' | 'running' | 'completed' | 'failed';
export type Priority = 'P0' | 'P1' | 'P2';
export type Complexity = 'low' | 'medium' | 'high';

// 技能接口
export interface Skill {
  id: string;
  name: string;
  version: string;
  execute(input: any): Promise<SkillResult>;
}

// 技能执行结果
export interface SkillResult {
  status: 'success' | 'error';
  data?: any;
  error?: string;
  duration?: number;
}

// 路由决策
export interface RoutingDecision {
  routeTo: string;
  reasoning: string;
  factors: Factor[];
  isOverride: boolean;
}

export interface Factor {
  name: string;
  value: any;
  weight: number;
  matchedRule: string;
}
```

**验收标准:**
- [ ] 类型定义完整
- [ ] TypeScript 编译通过

**预计工时:** 20 分钟
