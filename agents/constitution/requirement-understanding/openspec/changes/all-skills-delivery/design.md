## 5. 数据模型

### 5.1 技能执行记录

```typescript
interface SkillExecutionRecord {
  taskId: string;
  skillId: string;
  skillVersion: string;
  status: 'success' | 'error' | 'pending';
  input: any;
  output: any;
  duration: number; // ms
  timestamp: string;
  error?: string;
  metadata?: {
    phase: 'phase1' | 'phase2';
    priority: 'P0' | 'P1' | 'P2';
  };
}
```

### 5.2 技能状态

```typescript
interface SkillStatus {
  skillId: string;
  name: string;
  phase: 'phase1' | 'phase2';
  priority: 'P0' | 'P1' | 'P2';
  status: 'completed' | 'testing' | 'developing' | 'pending';
  testStatus?: {
    passed: boolean;
    testCount: number;
    coverage: number;
  };
  lastUpdated: string;
}

  timeout?: number;
  phase?: 'phase1' | 'phase2';
}

// 响应消息
interface SkillResponse {
  type: 'response';
  id: string;
  requestId: string;
  from: string;
  to: string;
  status: 'success' | 'error';
  result?: any;
  error?: string;
  duration?: number;
}
```
